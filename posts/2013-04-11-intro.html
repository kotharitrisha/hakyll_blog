module Calc where
import StackVM
import Parser
import qualified Data.Map as M
import qualified StackVM as S
import ExprT as E

reify :: ExprT -> ExprT
reify = id

--Defining Class Expr
class Expr a where
  mul :: a -> a -> a
  add :: a -> a -> a
  lit :: Integer -> a

--Defining instance of Expression for ExprT
instance Expr ExprT where
  mul = E.Mul
  add = E.Add
  lit = Lit

--Defining instance of Expression for Integers
instance Expr Integer where
  lit n  = n
  add = (+)
  mul = (*)
